SDK 示例代码

quickview-demo.js 
// 初始化 QuickView Link SDK
import { QuickViewLink, VideoConfig } from '@quickview/link-sdk';

// 配置参数
const config = new VideoConfig({
  latency: 10,        // 延迟目标 (ms)
  quality: 'auto',   // 自适应质量
  codec: 'h264',    // H.264 编码
  encryption: 'aes-256' // AES-256 加密
});

// 创建客户端实例
const client = new QuickViewLink(config);

// 连接摄像头
await client.connect({
  deviceId: 'camera-001',
  streamType: 'live'
});

// 监听视频流
client.on('frame', (frame) => {
  console.log(`Received frame: ${frame.timestamp}`);
  // 处理视频帧
});

// 断开连接
await client.disconnect();
